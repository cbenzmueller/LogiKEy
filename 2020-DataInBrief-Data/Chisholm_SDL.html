theory Chisholm_SDL imports SDL                          (*Christoph Benzm√ºller & Xavier Parent, 2019*)
begin     (*Unimportant*) nitpick_params [user_axioms,show_all,format=2] 

(*** Chisholm Example ***)
  consts go::σ tell::σ kill::σ
  abbreviation  "D1 ≡ ❙○<go>"  (*It ought to be that Jones goes to assist his neighbors.*)
  abbreviation  "D2w ≡ ❙○<go ❙→ tell>"  (*It ought to be that if Jones goes, then he tells them he is coming.*)
  abbreviation  "D2n ≡ go ❙→ ❙○<tell>"  
  abbreviation  "D3w ≡ ❙○<❙¬go ❙→ ❙¬tell>" (*If Jones doesn't go, then he ought not tell them he is coming.*)
  abbreviation  "D3n ≡ ❙¬go ❙→ ❙○<❙¬tell>" 
  abbreviation  "D4 ≡ ❙¬go" (*Jones doesn't go. (This is encoded as a locally valid statement.)*)

(*** Chisholm_A: All-wide scoping is leading to an inadequate, dependent set of the axioms.***)
 lemma "⌊(D1  ❙∧ D2w ❙∧ D3w) ❙→ D4⌋"  nitpick oops (*countermodel*)
 lemma "⌊(D1  ❙∧ D2w ❙∧ D4)  ❙→ D3w⌋"  sledgehammer by blast  (*proof*)
 lemma "⌊(D1  ❙∧ D3w ❙∧ D4)  ❙→ D2w⌋"  nitpick oops (*countermodel*)
 lemma "⌊(D2w ❙∧ D3w ❙∧ D4)  ❙→ D1⌋"  nitpick oops (*countermodel*)
 (* Consistency *)
 lemma "⌊(D1 ❙∧ D2w ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" nitpick [satisfy] oops (*Consistent? Yes*) 
 (* Queries *)
 lemma assumes "⌊(D1 ❙∧ D2w ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<❙¬tell>⌋⇩l" nitpick oops (*Should James not tell? No*) 
 lemma assumes "⌊(D1 ❙∧ D2w ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<tell>⌋⇩l"  using assms by blast (*Should J. tell? Yes*)
 lemma assumes "⌊(D1 ❙∧ D2w ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<kill>⌋⇩l"  nitpick oops (*Should James kill? No*)

(*** Chisholm_B: All-narrow scoping is leading to a inadequate, dependent set of the axioms.*)
 lemma "⌊(D1  ❙∧ D2n ❙∧ D3n) ❙→ D4⌋"   nitpick oops (*countermodel*)
 lemma "⌊(D1  ❙∧ D2n ❙∧ D4)  ❙→ D3n⌋"  nitpick oops  (*countermodel*)
 lemma "⌊(D1  ❙∧ D3n ❙∧ D4)  ❙→ D2n⌋"  sledgehammer by blast  (*proof*)
 lemma "⌊(D2n ❙∧ D3n ❙∧ D4)  ❙→ D1⌋"   nitpick oops (*countermodel*)
 (* Consistency *)
 lemma "⌊(D1 ❙∧ D2n ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" nitpick [satisfy] oops (*Consistent? Yes*) 
 (* Queries *)
 lemma assumes "⌊(D1 ❙∧ D2n ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<❙¬tell>⌋⇩l" using assms by smt (*Should J. not tell? Yes*) 
 lemma assumes "⌊(D1 ❙∧ D2n ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<tell>⌋⇩l"  nitpick oops (*Should James tell? No*)
 lemma assumes "⌊(D1 ❙∧ D2n ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<kill>⌋⇩l"  nitpick oops (*Should James kill? No*)

(*** Chisholm_C: Wide-narrow scoping is leading to an adequate, independence of the axioms.*)
 lemma "⌊(D1  ❙∧ D2w ❙∧ D3n) ❙→ D4⌋"  nitpick oops (*countermodel*)
 lemma "⌊(D1  ❙∧ D2w ❙∧ D4) ❙→ D3n⌋"  nitpick oops (*countermodel*)
 lemma "⌊(D1  ❙∧ D3n ❙∧ D4) ❙→ D2w⌋"  nitpick oops (*countermodel*)
 lemma "⌊(D2w ❙∧ D3n ❙∧ D4) ❙→ D1⌋"   nitpick oops (*countermodel*)
 (* Consistency *)
 lemma "⌊(D1 ❙∧ D2w ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" nitpick [satisfy] oops (*Consistent? No*) 
 (* Queries *)
 lemma assumes "⌊(D1 ❙∧ D2w ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<❙¬tell>⌋⇩l" using D assms by smt (*Shld J. not tell? Yes*) 
 lemma assumes "⌊(D1 ❙∧ D2w ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<tell>⌋⇩l"  using assms by blast (*Should J. tell? Yes*)
 lemma assumes "⌊(D1 ❙∧ D2w ❙∧ D3n)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<kill>⌋⇩l"  using D assms by blast (*Should J. kill? Yes*)

(*** Chisholm_D: Narrow-wide scoping is leading to a  inadequate, dependent set of the axioms.*)
 lemma "⌊(D1  ❙∧ D2n ❙∧ D3w) ❙→ D4⌋"   nitpick oops (*countermodel*)
 lemma "⌊(D1  ❙∧ D2n ❙∧ D4)  ❙→ D3w⌋"  by blast  (*proof*)
 lemma "⌊(D1  ❙∧ D3w ❙∧ D4)  ❙→ D2n⌋"  by blast  (*proof*)
 lemma "⌊(D2n ❙∧ D3w ❙∧ D4)  ❙→ D1⌋"   nitpick oops (*countermodel*)
 (* Consistency *)
 lemma "⌊(D1 ❙∧ D2n ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" nitpick [satisfy] oops (*Consistent? Yes*) 
 (* Queries *)
 lemma assumes "⌊(D1 ❙∧ D2n ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<❙¬tell>⌋⇩l" nitpick oops (*Should James not tell? No*) 
 lemma assumes "⌊(D1 ❙∧ D2n ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<tell>⌋⇩l"  nitpick oops (*Should James tell? No*)
 lemma assumes "⌊(D1 ❙∧ D2n ❙∧ D3w)⌋ ∧ ⌊D4⌋⇩l" shows "⌊❙○<kill>⌋⇩l"  nitpick oops (*Should James kill? No*)
